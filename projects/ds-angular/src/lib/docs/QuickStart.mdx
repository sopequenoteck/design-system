# Quick Start

Démarrez rapidement avec DS-Angular en 5 minutes.

---

## Premier composant

### 1. Importer un composant

```typescript
import { Component } from '@angular/core';
import { DsButton } from '@kksdev/ds-angular';

@Component({
  standalone: true,
  imports: [DsButton],
  template: `
    <ds-button variant="primary" (clicked)="onClick()">
      Mon premier bouton
    </ds-button>
  `
})
export class MyComponent {
  onClick() {
    console.log('Bouton cliqué !');
  }
}
```

## Exemple : Formulaire simple

```typescript
import { Component } from '@angular/core';
import { FormsModule } from '@angular/forms';
import { DsInputField, DsButton, DsCard } from '@kksdev/ds-angular';

@Component({
  standalone: true,
  imports: [FormsModule, DsInputField, DsButton, DsCard],
  template: `
    <ds-card>
      <ds-card-header>
        <h3>Connexion</h3>
      </ds-card-header>
      <ds-card-body>
        <ds-input-field
          label="Email"
          type="email"
          [(ngModel)]="email"
          placeholder="votre@email.com"
        />
        <ds-input-field
          label="Mot de passe"
          type="password"
          [(ngModel)]="password"
          style="margin-top: 16px"
        />
      </ds-card-body>
      <ds-card-footer>
        <ds-button variant="primary" [block]="true" (clicked)="login()">
          Se connecter
        </ds-button>
      </ds-card-footer>
    </ds-card>
  `
})
export class LoginComponent {
  email = '';
  password = '';

  login() {
    console.log('Login:', this.email);
  }
}
```

## Exemple : Liste avec actions

```typescript
import { Component } from '@angular/core';
import { DsTable, DsButton, DsChip, DsBadge } from '@kksdev/ds-angular';

@Component({
  standalone: true,
  imports: [DsTable, DsButton, DsChip, DsBadge],
  template: `
    <ds-table [data]="users" [columns]="columns">
      <ng-template #statusCell let-row>
        <ds-badge [variant]="row.active ? 'success' : 'error'">
          {{ row.active ? 'Actif' : 'Inactif' }}
        </ds-badge>
      </ng-template>
      <ng-template #actionsCell let-row>
        <ds-button size="sm" variant="ghost" (clicked)="edit(row)">
          Modifier
        </ds-button>
      </ng-template>
    </ds-table>
  `
})
export class UserListComponent {
  users = [
    { id: 1, name: 'Alice', email: 'alice@example.com', active: true },
    { id: 2, name: 'Bob', email: 'bob@example.com', active: false },
  ];

  columns = [
    { key: 'name', label: 'Nom' },
    { key: 'email', label: 'Email' },
    { key: 'active', label: 'Statut', template: 'statusCell' },
    { key: 'actions', label: '', template: 'actionsCell' },
  ];

  edit(user: any) {
    console.log('Edit:', user);
  }
}
```

## Exemple : Notifications

```typescript
import { Component, inject } from '@angular/core';
import { DsButton, DsToastService } from '@kksdev/ds-angular';

@Component({
  standalone: true,
  imports: [DsButton],
  template: `
    <ds-button variant="success" (clicked)="showSuccess()">
      Succès
    </ds-button>
    <ds-button variant="error" (clicked)="showError()">
      Erreur
    </ds-button>
  `
})
export class NotificationComponent {
  private toast = inject(DsToastService);

  showSuccess() {
    this.toast.success('Opération réussie !');
  }

  showError() {
    this.toast.error('Une erreur est survenue.');
  }
}
```

## Gestion des thèmes

### Changer de thème dynamiquement

```typescript
import { Component } from '@angular/core';
import { DsButton, DsSegmentedControl } from '@kksdev/ds-angular';

@Component({
  standalone: true,
  imports: [DsButton, DsSegmentedControl],
  template: `
    <ds-segmented-control
      [(ngModel)]="currentTheme"
      [options]="themeOptions"
      (ngModelChange)="changeTheme($event)"
    />
  `
})
export class ThemeSwitcherComponent {
  currentTheme = 'light';
  themeOptions = [
    { value: 'light', label: 'Clair' },
    { value: 'dark', label: 'Sombre' },
    { value: 'custom', label: 'Custom' },
  ];

  changeTheme(theme: string) {
    document.documentElement.className = `theme-${theme}`;
  }
}
```
