## 2. Formulaire d'inscription complet

Un formulaire multi-étapes avec validation, stepper et feedback visuel.

```typescript
import { Component, signal, computed } from '@angular/core';
import { FormBuilder, FormGroup, Validators, ReactiveFormsModule } from '@angular/forms';
import {
  DsCard,
  DsStepper,
  DsInputField,
  DsSelect,
  DsCheckbox,
  DsRadioGroup,
  DsButton,
  DsAlert,
  DsPasswordStrength,
  DsDatePicker,
  DsFileUpload
} from '@kksdev/ds-angular';

@Component({
  selector: 'app-registration',
  standalone: true,
  imports: [
    ReactiveFormsModule,
    DsCard, DsStepper, DsInputField, DsSelect, DsCheckbox,
    DsRadioGroup, DsButton, DsAlert, DsPasswordStrength,
    DsDatePicker, DsFileUpload
  ],
  template: `
    <ds-card variant="elevated" class="registration-card">
      <h1 slot="header">Créer un compte</h1>

      <!-- Stepper -->
      <ds-stepper
        [steps]="steps"
        [activeStep]="currentStep()"
        [linear]="true"
        (stepChange)="onStepChange($event)"
      />

      <!-- Step 1: Informations personnelles -->
      @if (currentStep() === 0) {
        <form [formGroup]="personalForm" class="form-section">
          <div class="form-row">
            <ds-input-field
              label="Prénom"
              formControlName="firstName"
              [required]="true"
              placeholder="Votre prénom"
            />
            <ds-input-field
              label="Nom"
              formControlName="lastName"
              [required]="true"
              placeholder="Votre nom"
            />
          </div>

          <ds-input-field
            label="Email"
            type="email"
            formControlName="email"
            [required]="true"
            placeholder="votre@email.com"
            [error]="emailError()"
          />

          <ds-date-picker
            label="Date de naissance"
            formControlName="birthDate"
            [maxDate]="maxBirthDate"
          />

          <ds-file-upload
            label="Photo de profil"
            accept="image/*"
            [maxSize]="5 * 1024 * 1024"
            (fileSelect)="onPhotoSelect($event)"
          />
        </form>
      }

      <!-- Step 2: Sécurité -->
      @if (currentStep() === 1) {
        <form [formGroup]="securityForm" class="form-section">
          <ds-input-field
            label="Mot de passe"
            type="password"
            formControlName="password"
            [required]="true"
            placeholder="Minimum 8 caractères"
          />

          <ds-password-strength
            [password]="securityForm.get('password')?.value"
            [showRequirements]="true"
          />

          <ds-input-field
            label="Confirmer le mot de passe"
            type="password"
            formControlName="confirmPassword"
            [required]="true"
            [error]="passwordMismatch() ? 'Les mots de passe ne correspondent pas' : ''"
          />

          <ds-checkbox formControlName="twoFactor">
            Activer l'authentification à deux facteurs
          </ds-checkbox>
        </form>
      }

      <!-- Step 3: Préférences -->
      @if (currentStep() === 2) {
        <form [formGroup]="preferencesForm" class="form-section">
          <ds-select
            label="Langue"
            formControlName="language"
            [options]="languageOptions"
            placeholder="Sélectionnez une langue"
          />

          <ds-radio-group
            label="Thème préféré"
            formControlName="theme"
            [options]="themeOptions"
            direction="horizontal"
          />

          <div class="checkbox-group">
            <h4>Notifications</h4>
            <ds-checkbox formControlName="emailNotifs">
              Notifications par email
            </ds-checkbox>
            <ds-checkbox formControlName="pushNotifs">
              Notifications push
            </ds-checkbox>
            <ds-checkbox formControlName="smsNotifs">
              Notifications SMS
            </ds-checkbox>
          </div>

          <ds-checkbox formControlName="terms" [required]="true">
            J'accepte les <a href="#">conditions d'utilisation</a>
          </ds-checkbox>
        </form>
      }

      <!-- Step 4: Confirmation -->
      @if (currentStep() === 3) {
        <div class="confirmation-section">
          <ds-alert type="success" [showIcon]="true">
            Votre compte a été créé avec succès !
          </ds-alert>

          <ds-card variant="outlined" class="summary-card">
            <h3>Récapitulatif</h3>
            <dl class="summary-list">
              <dt>Nom complet</dt>
              <dd>{{ fullName() }}</dd>
              <dt>Email</dt>
              <dd>{{ personalForm.get('email')?.value }}</dd>
              <dt>Langue</dt>
              <dd>{{ selectedLanguage() }}</dd>
              <dt>Thème</dt>
              <dd>{{ preferencesForm.get('theme')?.value }}</dd>
            </dl>
          </ds-card>
        </div>
      }

      <!-- Navigation -->
      <footer slot="footer" class="form-actions">
        @if (currentStep() > 0 && currentStep() < 3) {
          <ds-button variant="outline" (click)="previousStep()">
            Précédent
          </ds-button>
        }
        @if (currentStep() < 2) {
          <ds-button
            variant="primary"
            [disabled]="!canProceed()"
            (click)="nextStep()"
          >
            Suivant
          </ds-button>
        }
        @if (currentStep() === 2) {
          <ds-button
            variant="primary"
            [disabled]="!canSubmit()"
            (click)="submit()"
          >
            Créer mon compte
          </ds-button>
        }
        @if (currentStep() === 3) {
          <ds-button variant="primary" (click)="goToDashboard()">
            Accéder au dashboard
          </ds-button>
        }
      </footer>
    </ds-card>
  `,
  styles: [`
    .registration-card {
      max-width: 600px;
      margin: var(--space-8) auto;
    }

      flex-direction: column;
      gap: var(--space-4);
      padding: var(--space-6) 0;
    }
    .form-row {
      display: grid;
      grid-template-columns: 1fr 1fr;
      gap: var(--space-4);
    }

      display: flex;
      flex-direction: column;
      gap: var(--space-2);

      margin-top: var(--space-4);
    }
    .summary-list {
      display: grid;
      grid-template-columns: 1fr 2fr;
      gap: var(--space-2);
    }
    .summary-list dt {
      font-weight: 500;
      color: var(--text-secondary);
    }
  `]
```

```typescript
          [groups]="navGroups()"

          <ds-checkbox-list
            [items]="filterItems"
            size="sm"
            (selectionChange)="onFilterChange($event)"
          />
        </div>
      </aside>

      <!-- Contenu principal -->
      <main class="todo-content">
        <!-- Header -->
        <header class="content-header">
          <h1>{{ currentListName() }}</h1>
          <div class="header-actions">
            <ds-search-input
              placeholder="Rechercher une tâche..."
              [debounce]="300"
              (search)="onSearch($event)"
            />
            <ds-dropdown [items]="sortOptions" (itemSelect)="onSortChange($event)">
              <ds-button variant="outline" size="sm">
                Trier par
              </ds-button>
            </ds-dropdown>

          </div>
        </header>

        <!-- Liste de tâches groupées par date -->

          <ds-list
            variant="divided"
            [draggable]="true"
            [dragData]="filteredTasks()"

          >
            @for (group of groupedTasks(); track group.date) {
              <ds-list-group
                [title]="group.label"
                [count]="group.tasks.length"
                variant="collapsible"
                [expanded]="true"
              >
                @for (task of group.tasks; track task.id) {
                  <ds-list-item
                    [title]="task.title"
                    [checkable]="true"
                    [checked]="task.completed"
                    [completed]="task.completed"
                    indicator="priority"
                    [indicatorColor]="task.priority"
                    (checkedChange)="toggleTask(task, $event)"
                    (clicked)="openTaskDetail(task)"
                    cdkDrag
                  >
                    <!-- Tags inline -->
                    @for (tag of task.tags; track tag) {
                      <ds-chip inline size="sm" variant="outlined">
                        {{ tag }}
                      </ds-chip>
                    }

                    <!-- Date meta -->
                    @if (task.dueDate) {
                      <span meta [class.overdue]="isOverdue(task)">
                        {{ task.dueDate | date:'shortDate' }}
                      </span>
                    }

                    <!-- Actions au hover -->
                    <ds-dropdown actions [items]="taskActions" (itemSelect)="onTaskAction(task, $event)">
                      <ds-button variant="ghost" size="sm">
                        <fa-icon [icon]="faEllipsis" />
                      </ds-button>
                    </ds-dropdown>
                  </ds-list-item>
                }
              </ds-list-group>
            }
          </ds-list>
```

## 5. Tableau de données avancé

Tableau avec filtres, tri, sélection et pagination.

```typescript
import { Component, signal, computed } from '@angular/core';
import {
  DsCard,
  DsTable,
  DsSearchInput,
  DsSelect,
  DsButton,
  DsPagination,
  DsChip,
  DsBadge,
  DsDropdown,
  DsModal,
  DsSkeleton,
  DsEmpty,
  DsToast
} from '@kksdev/ds-angular';

@Component({
  selector: 'app-data-table',
  standalone: true,
  imports: [
    DsCard, DsTable, DsSearchInput, DsSelect, DsButton,
    DsPagination, DsChip, DsBadge, DsDropdown, DsModal,
    DsSkeleton, DsEmpty, DsToast
  ],
  template: `
    <ds-card variant="default">
      <header slot="header" class="table-header">
        <h2>Utilisateurs</h2>
        <ds-badge variant="info">{{ totalUsers() }} total</ds-badge>
      </header>

      <!-- Toolbar -->
      <div class="table-toolbar">
        <div class="toolbar-left">
          <ds-search-input
            placeholder="Rechercher..."
            [value]="searchQuery()"
            (search)="onSearch($event)"
          />

          <ds-select
            placeholder="Rôle"
            [options]="roleOptions"
            [value]="roleFilter()"
            [clearable]="true"
            (valueChange)="onRoleFilter($event)"
          />

          <ds-select
            placeholder="Statut"
            [options]="statusOptions"
            [value]="statusFilter()"
            [clearable]="true"
            (valueChange)="onStatusFilter($event)"
          />
        </div>

        <div class="toolbar-right">
          @if (selectedRows().length > 0) {
            <ds-chip variant="filled" color="primary">
              {{ selectedRows().length }} sélectionné(s)
            </ds-chip>
            <ds-dropdown [items]="bulkActions" (itemSelect)="onBulkAction($event)">
              <ds-button variant="outline">
                Actions groupées
              </ds-button>
            </ds-dropdown>
          }

          <ds-button variant="primary" (click)="openCreateModal()">
            <fa-icon [icon]="faPlus" /> Ajouter
          </ds-button>
        </div>
      </div>

      <!-- Active filters -->
      @if (hasActiveFilters()) {
        <div class="active-filters">
          @if (searchQuery()) {
            <ds-chip
              variant="outlined"
              [removable]="true"
              (remove)="clearSearch()"
            >
              Recherche: {{ searchQuery() }}
            </ds-chip>
          }
          @if (roleFilter()) {
            <ds-chip
              variant="outlined"
              [removable]="true"
              (remove)="clearRoleFilter()"
            >
              Rôle: {{ roleFilter() }}
            </ds-chip>
          }
          @if (statusFilter()) {
            <ds-chip
              variant="outlined"
              [removable]="true"
              (remove)="clearStatusFilter()"
            >
              Statut: {{ statusFilter() }}
            </ds-chip>
          }
          <ds-button variant="ghost" size="sm" (click)="clearAllFilters()">
            Effacer tout
          </ds-button>
        </div>
      }

      <!-- Table -->
      @if (loading()) {
        <ds-skeleton variant="table" [rows]="5" />
      } @else if (filteredUsers().length === 0) {
        <ds-empty
          title="Aucun résultat"
          description="Essayez de modifier vos filtres"
          [icon]="faUsers"
        />
      } @else {
        <ds-table
          [columns]="columns"
          [data]="paginatedUsers()"
          [selectable]="true"
          [selectedRows]="selectedRows()"
          variant="striped"
          stickyHeader
          (sortChange)="onSort($event)"
          (selectionChange)="onSelectionChange($event)"
          (rowClick)="onRowClick($event)"
        >
          <!-- Custom cell templates -->
          <ng-template dsTableCell="name" let-row>
            <div class="user-cell">
              <ds-avatar [name]="row.name" size="sm" />
              <div>
                <span class="user-name">{{ row.name }}</span>
                <span class="user-email">{{ row.email }}</span>
              </div>
            </div>
          </ng-template>

          <ng-template dsTableCell="role" let-row>
            <ds-badge [variant]="getRoleBadgeVariant(row.role)">
              {{ row.role }}
            </ds-badge>
          </ng-template>

          <ng-template dsTableCell="status" let-row>
            <ds-chip
              [variant]="row.status === 'active' ? 'filled' : 'outlined'"
              [color]="row.status === 'active' ? 'success' : 'default'"
              size="sm"
            >
              {{ row.status === 'active' ? 'Actif' : 'Inactif' }}
            </ds-chip>
          </ng-template>

          <ng-template dsTableCell="actions" let-row>
            <ds-dropdown [items]="rowActions" (itemSelect)="onRowAction(row, $event)">
              <ds-button variant="ghost" size="sm">
                <fa-icon [icon]="faEllipsisV" />
              </ds-button>
            </ds-dropdown>
          </ng-template>
        </ds-table>
      }

      <!-- Pagination -->
      <footer slot="footer" class="table-footer">
        <ds-select
          [options]="pageSizeOptions"
          [value]="pageSize()"
          (valueChange)="onPageSizeChange($event)"
          size="sm"
        />

        <ds-pagination
          [currentPage]="currentPage()"
          [totalPages]="totalPages()"
          [showFirstLast]="true"
          (pageChange)="onPageChange($event)"
        />
      </footer>
    </ds-card>
  `,
  styles: [`
    .table-header {
      display: flex;
      align-items: center;
      gap: var(--space-3);
    }
    .table-toolbar {
      display: flex;
      justify-content: space-between;
      align-items: center;
      padding: var(--space-4);
      border-bottom: 1px solid var(--border-color);
    }
    .toolbar-left, .toolbar-right {
      display: flex;
      gap: var(--space-3);
      align-items: center;
    }
    .active-filters {
      display: flex;
      gap: var(--space-2);
      padding: var(--space-3) var(--space-4);
      background: var(--background-secondary);
    }
    .user-cell {
      display: flex;
      align-items: center;
      gap: var(--space-3);
    }
    .user-name {
      display: block;
      font-weight: 500;
    }
    .user-email {
      display: block;
      font-size: var(--font-size-sm);
      color: var(--text-secondary);
    }
    .table-footer {
      display: flex;
      justify-content: space-between;
      align-items: center;
    }
  `]
})
export class DataTableComponent {
  columns = [
    { key: 'name', label: 'Utilisateur', sortable: true },
    { key: 'role', label: 'Rôle', sortable: true },
    { key: 'status', label: 'Statut', sortable: true },
```
